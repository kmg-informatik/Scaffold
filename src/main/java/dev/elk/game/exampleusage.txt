public static void main(String[] args) throws Exception {

        Shader fragment = new Shader("Assets/Shaders/fragment.glsl", Shader.ShaderType.FRAGMENT);
        Shader vertex = new Shader("Assets/Shaders/vertex.glsl", Shader.ShaderType.VERTEX);
        ShaderProgram program = new ShaderProgram();
        program.attachShaders(fragment, vertex);

        Window window = new Window("Hello!", Color.black);
        window.init();


        PipelineNode startNode = new PipelineNode(
                new TestScene(window),
                window);

        PipelineNode endNode = new PipelineNode(
                new TestSecondScene(window),
                window
        );

        PipelineNode endGame = new PipelineNode(
                new TestEndScene(window),
                window
        );

        startNode.addLinks(start ->{
            if (window.getCurrentFrameCount() > 100) return endNode;
            else return null;
        });

        endNode.addLinks(end->{
            if (window.getCurrentFrameCount()%20==0) return startNode;
            else return null;
        }, end->{
            if (window.getCurrentFrameCount() > 60*3) return endGame;
            else return end;
        });

        GamePipeline pipeline = new GamePipeline(window, startNode);
        pipeline.addNodes(startNode, endNode);
        pipeline.run();

        window.destroy();
}
